def getDist(cols):
    fig, ax = plt.subplots(2, 5, figsize = (20,10))
    idx = 0 

    for col in cols:
        temp_df = telco[col]

        if temp_df.dtype == np.int64 or temp_df.dtype == np.float64:
            temp_df.hist(ax = ax[int(idx/5), idx%5])
        else:
            pd.DataFrame(temp_df.value_counts()).plot(kind = 'bar', ax = ax[int(idx/5), idx%5])
        ax[int(idx/5), idx%5].title.set_text(col)
        idx += 1

    plt.tight_layout()
